/*๑ฎฑฒ-้ฟ แฃฒ๏ญ ๒ ชฏ

#define MTCTRL_RXCMD	0 // ๏งจๅกญ้ฅ ๋ฏฌแฎค๓ ง เดฑ๋ 
#define MTCTRL_PROC		1 // ๐ กฎ๒  า
#define MTCTRL_RDFRAM	2 // ๏ฟฝ๏ฟฝ้ฅ FRAM
#define MTCTRL_WRFRAM	3 // ่กฏ้ฒผ FRAM
#define MTCTRL_UARTTR	4 // เฆฐๆฅ ๏ฟฝ๏ฟฝART

#define FSMST_Q0		0 // ่กฏ๓ฑช อ
#define FSMST_Q1		1 
#define FSMST_Q2		2
#define FSMST_Q3		3 // ๏ฒฒแฎฎโกา
#define FSMST_Q4		4 // ่กฏ๓ฑช อ
#define FSMST_Q5		5 
#define FSMST_Q6		6
#define FSMST_Q7		7 // ๏ฒฒแฎฎโก
#define FSMST_Q8		8 // ่กฏ๓ฑช อ
#define FSMST_Q9		9 
#define FSMST_Q10		10
#define FSMST_Q11		11 // ๏ฒฒแฎฎโก
#define FSMST_QZ		12 // ๏ฒฒแฎฎโก

static uint32_t curraddrtm=0; // ๒ฅชณ๑ฅฅ เฏซ๏งฅ๎ฉฅ ๋กฐๆณช่static uint32_t nextaddrtm=0; // ๑ซฅค๓นฅฅ เฏซ๏งฅ๎ฉฅ ๋กฐๆณช่int ProcTm(uint8_t rsymb, uint8_t* wrsymb); // ้ณฅ๐ ถจ๏ฟฝ๏ฟฝ๒ฎฌ ๒  า

// เฆฐๆญฅ๎ฎ ๏ฟฝ๏ฟฝ๑ฒฎฟ๎ฉฟ แฃฒ๏ญ ๒  ช๏ฎฒ๐ฎซฟ อ
static char mtstatectrl = MTCTRL_RXCMD;   
// เฆฐๆญฅ๎ฎ ๏ฟฝ๏ฟฝ๑ฒฎฟ๎ฉฟ แฃฒ๏ญ ๒  า
static char fsmstate = FSMST_Q1;
	
void ProcessFsmTm(void)
{
	static int steptm; 
	static uint8_t rdsymbol;
	static uint8_t wrsymbol;
	static uint8_t aTrbuf[32];
    uint8_t* pRxbuf=0;
	int curstep=0;
	switch (mtstatectrl)
	{
		case MTCTRL_RXCMD: // ๏งจๅกญ้ฅ ่ก ๎กซ้ง ๋ฏฌแฎค๓		if(GetMessage(MSG_TM1STRT,(void*)&pRxbuf))
			{
				pRxbuf[lenfsm]=0;
				nextaddrtm=-1;
				steptm=-1;
				sscanf((char*)pRxbuf,"%x:%x",&nextaddrtm,&steptm);
				if(nextaddrtm>=0 && nextaddrtm<=0x7FF && steptm > 0)
				{// ่กฏ๓ฑช อ
					mtstatectrl = MTCTRL_RDFRAM;
					fsmstate = FSMST_Q1;
					ResetTimer(TM1_TIMER);
				}
				else
				{// ๑จญฒแซฑ้ธฅ๑ช ฟ ๏นจโซ  โกช๏ญ ๎ฅฅ
					strcpy((char*)aTrbuf,"Err> cmd error\n");
					mtstatectrl = MTCTRL_UARTTR;
					steptm = -1;
				}
			}
			break;
		case MTCTRL_PROC:
			curstep=ProcTm(rdsymbol,&wrsymbol);
			if(curstep==-1 || curstep == steptm)
			{// ๏ฒฒแฎฎใซ  เฎ ๑ฒฎฏ(-1) ้ฌจ ๋ฏซ้ธฅ๑ฒขณ ๐ ฃฎโ				sprintf((char*)aTrbuf,"TM1>stop addr=%03X sym='%c'\n",curraddrtm,wrsymbol);
				steptm = -1;
			}
			else
				sprintf((char*)aTrbuf,"TM1>step=%d addr=%03X sym='%c'\n",curstep,curraddrtm,wrsymbol);
			if(wrsymbol == rdsymbol) // ๆฒซ่กฑ้ญข๏ฌป ๑ฎขฏแฅ ๏ฟฝ๏ฟฝ๎กงแฐจ๑ผ ฏ๐ฎฏณ๑ช ฅ์				mtstatectrl = MTCTRL_UARTTR;
			else 
				mtstatectrl = MTCTRL_WRFRAM;
			break;
		case MTCTRL_RDFRAM:  
			curraddrtm=nextaddrtm; // เฆฐๆฃฎๅฉฌ ๋กฐๆณช๓ 			pRxbuf = (uint8_t*)i2cFRAM_rd(curraddrtm,1);
			if( (uint32_t)pRxbuf > 0)
			{//๏ฟฝ๏ฟฝๆฌ ๏ฅจ๏ฟฝแชฒ เฏค ๋กฐๆณช๏ฉ
				rdsymbol=pRxbuf[0];
				mtstatectrl = MTCTRL_PROC;
			}
			break;
		case MTCTRL_WRFRAM: //เฉธๆฌ ๏ฅจ๏ฟฝแชฒ
			if((uint32_t)i2cFRAM_wr(curraddrtm,&wrsymbol,1) > 0)
				mtstatectrl = MTCTRL_UARTTR;
			break;
		case MTCTRL_UARTTR:
			if(GetTimer(TM1_TIMER) > 1000) // 1000 msec
			{ // ๏งจๅกฅ์กข๐ฅฌฟ ้ณฅ๐ ถจ่				lenfsm = strlen((char*)aTrbuf);
				if( uart_transmit(aTrbuf,lenfsm) == 0 )
				{ // เฆฐๆฅ ูฌ ๐ฅงณ์ฝฒแฒ ้ณฅ๐ ถจ่กข ๒ฅฐฌ้ฎ ๋ 
					ResetTimer(TM1_TIMER);
					if(steptm == -1)
						mtstatectrl = MTCTRL_RXCMD;
					else
						mtstatectrl = MTCTRL_RDFRAM;
				}
			}
			break;

	}
}

int ProcTm(uint8_t rsymb,uint8_t* wrsymb)
{
	static int stepcnt=1;
	*wrsymb=rsymb;
	stepcnt++;
	switch(fsmstate)
	{
		case FSMST_Q0:
			stepcnt=1;
		case FSMST_Q1: // q1
			if(rsymb == '_' || rsymb == 'a' || rsymb == 'b') {
				// _,_,R
                // a,a,R
                // b,b,R
				nextaddrtm = curraddrtm + 1;
				fsmstate = FSMST_Q2;
			}
			break;
		case FSMST_Q2: // q2
			if(rsymb == 'a' || rsymb == 'b') {
				// a,a,R
                // b,b,R
				nextaddrtm = curraddrtm + 1;
			} else if(rsymb == '_') {
                // _,_,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q3;
			}
			break;
		case FSMST_Q3: // q3
			if(rsymb == 'b') {
				nextaddrtm = curraddrtm - 1;
			} else if(rsymb == 'a') {
				*wrsymb = '*'; // a,*,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q4;
			} else if(rsymb == '_') {
				// _,_,S
				fsmstate = FSMST_QZ;
			}
			break;
		case FSMST_Q4: // q4
			if(rsymb == 'b') {
				nextaddrtm = curraddrtm - 1;
			} else if(rsymb == 'a') {
				*wrsymb = '*'; // a,*,R
				nextaddrtm = curraddrtm + 1;
				fsmstate = FSMST_Q5;
			} else if(rsymb == '_') {
                // _,_,R
				nextaddrtm = curraddrtm + 1;
				fsmstate = FSMST_Q5;
			}
			break;
		case FSMST_Q5: // q5
			if(rsymb == 'b' || rsymb == '*') {
				// b,b,R
				nextaddrtm = curraddrtm + 1;
			} else if(rsymb == '_') {
				*wrsymb = '>'; // _,>,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q6;
			}
			break;
		case FSMST_Q6: // q6
			if(rsymb == 'a' || rsymb == 'b' || rsymb == '*') {
				// a,a,L
                // b,b,L
                // *,*,L
				nextaddrtm = curraddrtm - 1;
			} else if(rsymb == '_') {
				// _,_,R
				nextaddrtm = curraddrtm + 1;
                fsmstate = FSMST_Q7;
			}
			break;
		case FSMST_Q7: // q7
			if(rsymb == '*') {
				*wrsymb = '_'; // *,_,R
                nextaddrtm = curraddrtm + 1;
			} else if(rsymb == 'a') {
				*wrsymb = '_'; // a,_,R
				nextaddrtm = curraddrtm + 1;
				fsmstate = FSMST_Q8;
			} else if(rsymb == 'b') {
				*wrsymb = '_'; // b,_,R
				nextaddrtm = curraddrtm + 1;
				fsmstate = FSMST_Q10;
			} else if(rsymb == '>') {
				*wrsymb = '_'; // >,_,S
				fsmstate = FSMST_QZ;
			}
			break;
		case FSMST_Q8: // q8
			if(rsymb == 'a' || rsymb == 'b' || rsymb == '*' || rsymb == '>') {
				// a,a,R
                // b,b,R
                // *,*,R
                // >,>,R
				nextaddrtm = curraddrtm + 1;
			} else if(rsymb == '_') {
				*wrsymb = 'a'; // _,a,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q9;
			}
			break;
		case FSMST_Q9: // q9
			if(rsymb == 'a' || rsymb == 'b') {
				// a,a,L
                // b,b,L
				nextaddrtm = curraddrtm - 1;
			} else if(rsymb == '>') {
				// >,>,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q6;
			}
			break;
		case FSMST_Q10: // q10
			if(rsymb == 'a' || rsymb == 'b' || rsymb == '*' || rsymb == '>') {
				// a,a,R
                // b,b,R
                // *,*,R
                // >,>,R
				nextaddrtm = curraddrtm + 1;
			} else if(rsymb == '_') {
				*wrsymb = 'b'; // _,b,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q11;
			}
			break;
		case FSMST_Q11: // q11
			if(rsymb == 'a' || rsymb == 'b') {
				// a,a,L
                // b,b,L
				nextaddrtm = curraddrtm - 1;
			} else if(rsymb == '>') {
				// >,>,L
				nextaddrtm = curraddrtm - 1;
				fsmstate = FSMST_Q6;
			}
			break;
		case FSMST_QZ: // qz (final state)
			stepcnt = -1; // Halt
			break;
	}
	return stepcnt;
}
