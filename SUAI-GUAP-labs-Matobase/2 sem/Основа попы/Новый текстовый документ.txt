#include<iostream>
#include<locale>
using namespace std;

void remove_char(char* text, char ch, bool &flag_begin,bool &flag_end, int &flag_len) {
    int len = 0;
    for(int i = 0; text[i]; i++){
        len++;
    }
    bool find_from_begin = false;
    bool find_from_end = false;
    for (int i = 0; i <= len; i++) {
        if (text[i] == ch ) {
            for (int j = i; j <= (len - 1); j++) {
                text[j] = text[j + 1];
            }
            len--;
            find_from_begin = true;
            break;
        }
    }
    if (find_from_begin == true) {
        for (int i = len; i >= 0; i--) {
            if (text[i] == ch ) {
                for (int j = i; j <= (len - 1); j++) {
                    text[j] = text[j + 1];
                }
                len--;
                find_from_end = true;
                break;
            }
        }
    }
	flag_begin = find_from_begin;
	flag_end = find_from_end;
	flag_len = len;
}
